133 static int get_pte_idx(const char *label)
134 {
135  int i;
136  char atom[3];
137 
138  /* zap string */
139  atom[0] = (char) 0;
140  atom[1] = (char) 0;
141  atom[2] = (char) 0;
142  /* if we don't have a null-pointer, there must be at least two
143  * chars, which is all we need. we convert to the capitalization
144  * convention of the table above during assignment. */
145  if (label != NULL) {
146  atom[0] = (char) toupper((int) label[0]);
147  atom[1] = (char) tolower((int) label[1]);
148  }
149  /* discard numbers in atom label */
150  if (isdigit(atom[1])) atom[1] = (char) 0;
151 
152  for (i=0; i < nr_pte_entries; ++i) {
153  if ( (pte_label[i][0] == atom[0])
154  && (pte_label[i][1] == atom[1]) ) return i;
155  }
156 
157  return 0;
158 }

160 static int get_pte_idx_from_string(const char *label) {
161  int i, ind;
162  char atom[3];
163 
164  if (label != NULL) {
165  /* zap string */
166  atom[0] = atom[1] = atom[2] = '\0';
167 
168  for (ind=0,i=0; (ind<2) && (label[i]!='\0'); i++) {
169  if (label[i] != ' ') {
170  atom[ind] = toupper(label[i]);
171  ind++;
172  }
173  }
174 
175  if (ind < 1)
176  return 0; /* no non-whitespace characters */
177 
178  for (i=0; i < nr_pte_entries; ++i) {
179  if ((toupper(pte_label[i][0]) == atom[0]) && (toupper(pte_label[i][1]) == atom[1]))
180  return i;
181  }
182  }
183 
184  return 0;
185 }
