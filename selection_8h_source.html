804  template<class F>
805  void split(std::vector<Selection>& parts, F callback){
806  using namespace std;
807  parts.clear();
808  using Ret = decltype(callback(*this,0));
809  // Map
810  map<Ret,vector<int> > m;
811  for(int i=0; i<size(); ++i){
812  m[callback(*this,i)].push_back(index(i));
813  }
814  // Create selections
815  typename map<Ret,vector<int> >::iterator it;
816  for(it=m.begin();it!=m.end();it++){
817  parts.emplace_back(*system, it->second.begin(), it->second.end());
818  }
819  }
